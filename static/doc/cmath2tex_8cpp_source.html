138 std::string CMath2Tex::translate(const std::vector<std::string>& tokens)
139 {
140  std::vector<Lexeme> toperators;
141  std::vector<Lexeme> lexemes;
142  LexemeTree tr;
143  int level = 0;
144  int pos = 0;
145  //------------------------------------------------------------------
146  for (auto& t : tokens)
147  {
148  Lexeme l(t, pos);
149  do
150  {
151  if (l.type() == LexemeLibrary::bracketl) {
152  ++level;
153  tr.save_parenthesis_pos(l);
154  break;
155  }
156  if (l.type() == LexemeLibrary::bracketr) {
157  --level;
158  tr.save_parenthesis_pos(l);
159  break;
160  }
161 
162  if (LexemeLibrary::is_toperator(l.type()))
163  {
164  l.update_priority(level);
165  toperators.push_back(l);
166  }
167  else
168  {
169  lexemes.push_back(l);
170  }
171  }while(false);
172  ++pos;
173  }
174 
175  // sort operators by priority
176  std::sort(toperators.rbegin(), toperators.rend(),
177  [&](const Lexeme& l1, const Lexeme&l2) -> bool {
178  return l1.priority() < l2.priority();
179  });
180  GLogger::instance().logTrace("Operation list (sort by priority):"_i18n);
181  for(auto& lex : toperators)
182  {
183  GLogger::instance().logTrace("\t", lex.lexeme(), " with priority: "_i18n, lex.priority(), " and pos: "_i18n, lex.pos());
184  }
185  GLogger::instance().logTrace("Other lexemes list (sort by position):"_i18n);
186  for(auto& lex : lexemes)
187  {
188  GLogger::instance().logTrace("\t", lex.lexeme(), " with pos: "_i18n, lex.pos());
189  };
190 
191  // I pass - fill with transform operators
192  for(auto& op: toperators)
193  {
194  tr.insert(op);
195  }
196  GLogger::instance().logTrace("I. Operation tree (sort by position):"_i18n);
197  tr.output();
198  GLogger::instance().logDebug(tr.display());
199 
200  // II pass - fill with lexemes
201  for(auto& lex: lexemes)
202  {
203  tr.insert(lex);
204  }
205  GLogger::instance().logTrace("II. Final tree (sort by position):"_i18n);
206  tr.output();
207  GLogger::instance().logDebug(tr.display());
208  //
209  GLogger::instance().logTrace("Appling transformations:"_i18n);
210  return tr.transform(); // apply transformation
211 }
